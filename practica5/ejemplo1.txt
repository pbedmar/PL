principal ()
{
    inicio_var
        entero   n, curr ;
        lista de entero pe, pe2;
        entero tam, elem;
    fin_var 

    imprimir "introduce numero : ";
    leer n;
    imprimir " ", n, " == ";
    curr = 2;
    
    mientras (curr <= n) {
        inicio_var
            entero d;
        fin_var

        d = n/curr;

        si(d*curr == n) {
            imprimir " * ", curr;
            n = n / curr;
        } en otro caso {
            curr = curr+1 ;
        }
    }

    imprimir "\n";

    pe = pe++(6)@(0);
    pe = pe++(-2)@(0);
    pe = pe++(3)@(0);
    pe = pe++(7)@(#pe);
    $pe;
    pe>>;
    pe>>;
    pe<<;

    tam = #pe;
    elem = ?pe;
    elem = pe@1;
    pe = pe--1;
    pe = pe%(#pe-1);

    pe2 = pe2++(14)@(0);
    pe2 = pe2**pe;

    pe = pe+3;
    pe2 = 4+pe2;
    pe = pe-1;

    pe = pe*2;
    pe2 = 7*pe2;
    pe2 = pe2/3;
}
